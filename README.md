        if (lc === 'remote' && !job.remote) return false;
        if (!job.remote && lc !== 'remote' && !job.location.toLowerCase().includes(lc)) return false;

      }
      if (maxHoursVal) {
        if (job.hoursPerWeek > Number(maxHoursVal)) return false;
      }
      return true;
    }
